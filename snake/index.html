Get Started 

# Snake struct example
struct Person:
    name: str;
    age: int;
    height: float;

def greet_person(p: Person) -> str:
    return f"Hello, {p.name}! You are {p.age}."

person: Person = Person("Kate", 25, 1.65)
print(greet_person(person))
